create table test(
    id int unique,
    name char(50),
    grade float unique,
    sex char(1),
    primary key (id)
);
insert into test values(1, "bob", 1.2, "f");
insert into test values(2, "alice", 1.2, "m");
insert into test values(3, "charlie", 1.2, "f");
select * from test;
select * from test where id <> 1;
select * from test where id > 1;
create index f on test(grade);
delete from test where id <> 1;
select * from test where grade=1.2 and sex <> "m";
select * from test where sex="f";
drop index f;
drop table test;
